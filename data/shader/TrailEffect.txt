#include "ConstantHeader.txt"
struct VS_in
{
	float3 p : POSITION;
	float3 n : NORMAL;
	float4 c : COLOR;
	float2 t : TEXCOORD;
};
struct VS_out
{
	float4 p : SV_POSITION;
	float3 n : NORMAL;
	float4 c : COLOR0;
	float2 t : TEXCOORD0;
};

VS_out VS(VS_in input)
{
	VS_out output = (VS_out)0;

	float4 vWorld = float4(input.p, 1.0f);
	//float4 vView = mul(vWorld, g_matView);
	//float4 vProj = mul(vView, g_matProj);
	float4 vProj = mul(vWorld, g_matComb); // 위의 세 행렬을 곱한 행렬

	output.p = vProj;
	output.n = input.n;
	output.c = input.c;
	output.t = input.t;

	return output;
}

float4 PS(VS_out input) : SV_Target
{
	float4 vColor = g_txTexA.Sample(g_SampleWrap , input.t);
	if(vColor.r < 0.1)
		discard;
	return vColor * input.c;
}

float4 COLOR_PS(VS_out input) : SV_Target
{
	return input.c;
}
